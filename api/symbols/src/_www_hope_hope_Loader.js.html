 81 	/**
 82 		Initialize the loader.
 83 		Called automatically at end of Loader.js. 
 84 	*/
 85 	initialize : function() {
 86 		var documentPath = (""+window.location).toLocation().fullpath,
 87 			basePath = documentPath
 88 		;
 89 
 90 		// path to the document
 91 		Loader.Paths.document = documentPath;
 92 
 93 		// NOTE: assumes that this document is at the top-level for its 'app'
 94 		Loader.Paths.app = documentPath;
 95 
 96 		// At this point the loader path will be the src of the last script in the document
 97 		// Figure out the base path relative to the URL of the loader file.
 98 		var loaderScript = Loader.getLastScript(),
 99 			loaderUrl = Loader.loaderUrl = loaderScript.src,
100 			loaderPath = loaderUrl.toLocation().fullpath,
101 			loaderFileName = loaderUrl.toLocation().file,
102 			basePath = (loaderPath + "/../").toLocation().fullpath
103 		;
104 
105 		//
106 		// pre-determine a few package paths
107 		//
108 
109 		// base of peers of hope system
110 		Loader.Paths.base = basePath;
111 
112 		// cache directory (for pre-compiled scripts)
113 		Loader.Paths.cache = basePath + "cache/";
114 
115 		// test directory
116 		Loader.Paths.test = basePath + "test/";
117 
118 		// base of hope system
119 		Loader.Paths.hope = loaderPath;
120 
121 		// library directory
122 		Loader.Paths.library = loaderPath + "library/";
123 
124 
125 		// any inlined script in the Loader script element should be executed
126 		//	after the loader finishes loading.
127 		var loaderCallback = loaderScript.innerHTML;
128 		if (typeof loaderCallback == "string") loaderCallback = new Function(loaderCallback);
129 
130 		// if there is a 'packages' attribute of the loader script,
131 		//	load those packages after the hope package finshes loading
132 		//	(and call the loaderCallback once those packages have been loaded)
133 		var packages = loaderScript.getAttribute("packages");
134 		if (packages) {
135 			packages = packages.split(/\s*,\s*/);
136 			Loader.whenLoaded("package:hope", function() {
137 				Loader.loadPackages(packages, loaderCallback);
138 			});
139 		}
140 		// otherwise just execute the loaderCallback when the hope package has been loaded
141 		else {
142 			if (loaderCallback) {
143 				Loader.whenLoaded("package:hope", loaderCallback);
144 			}
145 		}
146 
147 		// set things up so we mark all inlined script tags as loaded
148 		Loader.whenLoaded("document", Loader.markPageElementsAsLoaded);
149 
150 		// If we are not loading from cache,
151 		//	load the hope package now with a bootstrap package loader.
152 		//	(The hope package will be initialized as a full package later.)
153 		if (loaderFileName != Loader.loaderCacheFileName) {
154 			Loader.loadHopePackage();
155 		}
156 	},

255 	/** Add a method to be called when something finishes loading.
256 		@param {String} it A URL or a conceptual name like "document" or "package:hope".
257 		@param {Function} Function or string to eval when it is loaded.
258 	*/
259 	whenLoaded : function(it, callback) {
260 		if (it == document) it = "document";
261 		if (typeof callback == "string") callback = new Function(callback);
262 
263 		// if it has already been loaded, just execute the callback
264 		if (Loader.isLoaded(it)) {
265 			callback();
266 		} else {
267 			if (!Loader.LoadCallbacks[it]) Loader.LoadCallbacks[it] = [];
268 			Loader.LoadCallbacks[it].push(callback);
269 		}

272 	/** Execute the on load callback(s) for something and mark it as loaded.
273 		@param {String} it A URL or a conceptual name like "document" or "package:hope".
274 		@param [loadResult] The thing that was instantiated from this load 
275 			(eg: the Package instance).
276 	*/
277 	onload : function(it, loadResult) {
278 		Loader._debug(".onload(",it,",",loadResult,")");
279 		// mark the thing as loaded
280 		Loader.setLoadResult(it, loadResult);
281 
282 		// and execute any load callbacks
283 		var callbacks = Loader.LoadCallbacks[it];
284 		if (callbacks) {;
285 			//console.group("executing load callbacks for "+it);
286 			callbacks.forEach(function(callback) {
287 				//console.warn("executing callbacks:\n"+callback);
288 				callback();
289 			});
290 			//console.groupEnd();
291 
292 			// clear the list of callbacks on the off-chance that our "onload" is fired again
293 			delete Loader.LoadCallbacks[it];
294 		}

297 	/** Load a URL as text.
298 		@note Will load the same url more than once.
299 		@param {String} url Url to load.
300 		@param {Boolean} [defer] (default: true) If true, we will defer the load (eg: asynchronous).
301 		@param {Function} [callback] Method to execute upon successful load.
302 		@param {Function} [errback] Method to execute if load fails.
303 		@param {String} [errHint] Hint to pass to the errback function 
304 			(eg: what you were doing that caused the load).
305 		@returns {String} If `defer == true`, returns contents of file as text.
306 						  If `defer == false`, returns null.
307 	*/
308 	loadText : function (url, defer, callback, errback, errHint) {
309 		url = Loader.absoluteUrl(url);
310 		var request = new XMLHttpRequest();
311 		var callbackWhenDone = function () {
312 			// wait until 'Completed' if a synchronous call
313 			if (request.readyState != 4) return;
314 
315 			// error state
316 			if (request.status < 200 || request.status > 300) {
317 				if (errback) return errback(url);
318 				if (!errHint) errHint = "Loader.loadText('"+url+"'): ";
319 				throw errHint + ": Error loading file ("+request.status+")";
320 			}
321 			var text = request.responseText;
322 			if (callback) return callback(text, request);
323 			return text;
324 		}
325 
326 		if (defer) request.onreadystatechange = callbackWhenDone;
327 		request.open("GET", url, defer==true);
328 		request.send();
329 		if (!defer) return callbackWhenDone();
330 	},

394 		<pre>
395 		NOTE: assumes the loadMethod:
396 				- throws an exception if it couldn't load the file
397 				- does NOT do any checking to see if that file is already loaded
398 				- takes the following arguments:
399 	
400 					function loadSomething(url, defer, callback, errback)
401 			where
402 						url = any valid url
403 						defer = true means we don't block waiting for this to finish
404 							- note that 'defer' may have penalties (such as making debugging harder)
405 							- note that each item loader can have its default defer()
406 						callback to execute when loaded, as:  callback(result, url)
407 						errback to execute on exception, as:  errback(error, url)
408 							- if errback (re)throws an error, that stops subsequent loading
409 							- if errback swallows the error, subsequent loading continues
410 			and returns
411 						if deferring, Loader.LOADING
412 						if not deferring, actual file contents from the load
413 		</pre>
414 	*/
415 	makeLoadable : function(info) {
416 		var loadOne = "load" + info.type,
417 			loadMany = "load" + (info.plural ? info.plural : info.type + "s")
418 		;
419 
420 		if (info.defer == true) {
421 			Loader[loadOne] = info.loadOne = 
422 			/** @ignore */
423 			function loadOne(url, callback, errback) {
424 				url = (info.getUrl ? info.getUrl(url) : Loader.absoluteUrl(url));
425 				try {
426 					var result = Loader.Loaded[url] || info.load(url, callback, errback);
427 					return (Loader.Loaded[url] = result);
428 				} catch (e) {
429 					Loader.Loaded[url] = Loader.LOAD_ERROR;
430 					if (errback) return errback(e, url);
431 					throw e;
432 				}
433 			}
434 
435 			Loader[loadMany] = info.loadMany = 
436 			/** @ignore */
437 			function loadMany(urls, callback, errback) {
438 				var results = {},
439 					loadNext = Loader._makeLoadingQueue(urls, info.loadOne, callback, errback, results)
440 				;
441 				loadNext();
442 				return results;
443 			}
444 		} else {
445 
446 			// NOTE: assumes:	- url is already an absolute url
447 			//					- callback takes 	(results, url)
448 			//					- errback takes 	(exception, url)
449 			
450 			Loader[loadOne] = info.loadOne = 
451 			/** @ignore */
452 			function loadOne(url, callback, errback) {
453 				url = (info.getUrl ? info.getUrl(url) : Loader.absoluteUrl(url));
454 				try {
455 					var result = Loader.Loaded[url] || info.load(url);
456 					if (callback) callback(result, url);
457 					return (Loader.Loaded[url] = result);
458 				} catch (e) {
459 					Loader.Loaded[url] = Loader.LOAD_ERROR;
460 					if (errback) return errback(e, url);
461 					throw e;
462 				}
463 			}
464 
465 			
466 			Loader[loadMany] = info.loadMany = 
467 			/** @ignore */
468 			function loadMany(urls, callback, errback) {
469 				var results = {};
470 				Array.forEach(urls, function(url) {
471 					try {
472 						results[url] = info.loadOne(url);
473 					} catch (e) {
474 						if (errback)  	errback(e, url);
475 						else			throw e;
476 					}
477 				});
478 				if (callback) callback();
479 				return results;
480 			}
481 		}
482 
483 		// now store the info to call this loader under each extension passed in
484 		if (info.extensions) {
485 			if (typeof info.extensions == "string") info.extensions = [info.extensions];
486 			info.extensions.forEach(function(extension) {
487 				Loader.ExtensionMap[extension] = info;
488 			});
489 		}

492 	/** Create a function to load each of the URLs in turn and call callback when done.
493 		<pre>
494 		 EITHER:  urls is a list of strings and loadMethod is a function to load each one
495 	  	     OR:	urls is a list of [url, loadMethod] and loadMethod is ignored
496 	  	 </pre>
497 	*/
498 	_makeLoadingQueue : function(urls, loadMethod, callback, errback, results) {
499 //console.info("making loading queue with ",urls);
500 		// if the individual load fails, call the errback passed in
501 		//	if that does not (re)throw, keep going
502 		/** @ignore */
503 		function loaderErrback(error, url) {
504 //console.error("in loaderErrback", error);
505 			Loader._error("Error loading url '"+url+"'");
506 			if (errback) {
507 				errback(error, url);
508 				loadNext();
509 			} else {
510 				throw error;
511 			}
512 		}
513 
514 		var index = -1;
515 		/** @ignore */
516 		function loadNext() {
517 			// if we're at the end of the list, call the callback and we're done
518 			if (++index >= urls.length) {
519 //console.warn("loaded last item, calling :"+callback);
520 				if (callback) callback(results);
521 				return;
522 			}
523 			var url = urls[index];
524 			if (typeof url != "string") {
525 				loadMethod = url[1];
526 				url = url[0];
527 			}
528 			// if the url was already loaded, just go on to the next one
529 			if (Loader.isLoaded(url)) {
530 //console.warn(url + " is loaded already loaded - skipping to next");
531 				return loadNext();
532 			}
533 
534 //console.warn("loading ",url," from queue");
535 			Loader.Loaded[url] = results[url] = loadMethod(url, loadNext, loaderErrback);
536 		}
537 		return loadNext;
538 	},

542 	/** Load a bunch of files ONLY ONCE according to their extension.
543 		@returns {Boolean} `true` if everything was already loaded or has been loaded synchronously, or
544 						<br/>`false` if at least one thing could not be loaded synchronously
545 	
546 		@sideeffect Sets Loader.loadFilesResults to a map of {url:<load results>} of the urls.
547 	
548 		<pre>
549 		If callback is defined, only calls that when ALL are loaded as:
550 				callback(resultsMap)
551 		If errback is defined, calls that when things that can't be loaded is loaded as:
552 				errback(error, <urlThatFailed>, urls)
553 			If the errback (re)throws an error, stops loading process.
554 			If errback swallows error, keeps loading the next things.
555 		</pre>
556 	*/
557 	loadFiles : function(urls, callback, errback, skipUnknownExtensions) {
558 		urls = Loader.absoluteUrls(urls);
559 		var results = Loader.loadFilesResults = {},
560 			asyncLoads = []
561 		;
562 		urls.forEach(function(url) {
563 			// skip things that have already been loaded
564 			if (Loader.Loaded[url]) {
565 				results[url] = Loader.Loaded[url];
566 				return;
567 			}
568 
569 			try {
570 				var extension = url.toLocation().extension,
571 					info = Loader.ExtensionMap[extension]
572 				;
573 				if (!info && skipUnknownExtensions != true) {
574 					results[url] = Loader.Loaded[url] = Loader.loadText(url);
575 
576 				} else if (info.defer != true) {
577 					results[url] = info.loadOne(url);
578 
579 				} else {
580 					asyncLoads.push([url, info.loadOne]);
581 					Loader.Loaded[url] = Loader.LOADING;
582 				}
583 			} catch (e) {
584 				// If errback is defined, call it
585 				//		If it (re)throws an exception, we'll error out of loadFiles() entirely
586 				if (errback) errback(e, url);
587 				// if errback is note defined, throw the error
588 				else		 throw e;
589 			}
590 		});
591 
592 		// if there is nothing to load asynchronously, we're done!
593 		if (asyncLoads.length == 0) {
594 			if (callback) callback(results, urls);
595 			return true;
596 		}
597 
598 		// load asynchronous things one at a time
599 		var loadNext = Loader._makeLoadingQueue(asyncLoads, null, callback, errback, results);
600 		loadNext();
601 		return false;		// signal that not everything was loaded
602 	},

658 				});
659 			}

661 				if (deps.indexOf(thing) == -1) deps.push(thing);
662 				Loader.Things[thing].forEach(function(it) {
663 					if (deps.indexOf(it) == -1) deps.push(it);
664 				});
665 			} else {
666 				thing = Loader.absoluteUrl(thing);
667 				if (deps.indexOf(thing) == -1) deps.push(thing);
668 			}
669 		});
670 		return deps;

676 	/** Convert a special characters in URL paths.
677 		Converts `.`, `..`, and `//`.
678 		@param {String} url URL to convert.
679 		@returns {String} URL with special characters converted.
680 	*/
681 	normalizeUrl : function(url) {
682 		// if we were passed a manifest entry, it will have a "url" property
683 		if (typeof url == "object" && url.url) url = url.url;
684 		// TOWARN
685 
686 		if (url.indexOf(".") == -1 && url.indexOf("//") == -1) return url;
687 		url = url.replace(/\/\.\/(\.\/)*/g,"/").replace(/([^:])\/\/+/g, "$1/");
688 		url = url.split("/");
689 		for (var i = 0; i < url.length; ) {
690 			if (url[i] == "..") 	url.splice(--i, 2);
691 			else					i++;
692 		}
693 		return url.join("/");
694 	},

713 		URL can have named paths in it, or can be a relative URL, or can be an absolute URL.
714 		@param {String} url Url to convert to an absolute URL.
715 		@param {String} [base] Base path to use if `url` is not already an absolute url.
716 				Defaults to `{document}`.
717 	*/
718 	absoluteUrl : function(url, base) {
719 		// if we were passed a manifest entry, it will have a "url" property
720 		if (typeof url == "object") {
721 			if (url.url) 				url = url.url;
722 			else if (url.getAttribute) 	url = url.getAttribute("url")
723 		}
724 
725 		var cacheName = url + base;
726 		// if we've already munged at this one before, return it
727 		if (Loader._absoluteUrlCache[cacheName])
728 			return Loader._absoluteUrlCache[cacheName];
729 
730 		url = Loader.expandNamedPath(url);
731 
732 		if (url.indexOf("http:") != 0 && url.indexOf("file:") != 0) {
733 			base = Loader.expandNamedPath(base || "{document}");
734 
735 			if (url.charAt(0) == "/") {
736 				url = base.toLocation().prefix + url;
737 			} else {
738 				url = base.toLocation().fullpath + url;
739 			}
740 		}
741 
742 		return (Loader._absoluteUrlCache[cacheName] = Loader.normalizeUrl(url));
743 	},

864 /**
865 	Breaks a URL up into pieces (much like window.location).
866 	@example var loc = new Location('http://www.server.com:81/path/to/a/file.xml?p1=v1&p2=v2#hash');
867 	@class
868 	@constructor
869 	@param {String} url URL to convert into a location object.
870 */
871 function Location(url) {
872 	if (typeof url != "string") throw "Location must be initialized with a string: "+url;
873 
874 	if (Location.Cache[url]) {
875 		this.match = Location.Cache[url].match;
876 	} else {
877 		var normalizedUrl = Loader.normalizeUrl(url);
878 		this.match = Location.urlParser.exec(normalizedUrl);
879 		if (!this.match) throw "Location not understood: "+url;
880 		// remember both under the normalized URL and the original one passed in
881 		Location.Cache[url] = Location.Cache[normalizedUrl] = this;
882 	}
883 }
